#include <iostream>
using namespace std;
template <typename type>
class date
{
	type dd, mm, yy;
public:
	date(type d = 0, type m = 0, type y = 0);
	void operator+(date& obj);
	void operator-(date& obj);
	void operator=(date& obj);
	void operator++();														// Prefix
	void operator--();														
	void operator++(type);													// Postfix
	void operator--(type);
	int operator<(date& obj);
	int operator>(date& obj);
	int operator<=(date& obj);
	int operator>=(date& obj);
	void output();
};
template <typename type>
 date<type>::date(type d , type m , type y )
{
	dd = d;
	mm = m;
	yy = y;
}
template<typename t>
void date<t>::operator+(date& obj)
{
	this->dd += obj.dd;
	this->mm += obj.mm;
	this->yy += obj.yy;
}
template<typename t>
void date<t>::operator-(date& obj)
{
	this->dd -= obj.dd;
	this->mm -= obj.mm;
	this->yy -= obj.yy;
}
template<typename t>
void date<t>::operator=(date& obj)
{
	dd = obj.dd;
	mm = obj.mm;
	yy = obj.yy;
}
template<typename t>												// prefix 
void date<t>::operator++()
{
	++dd;
	++mm;
	++yy;
}
template<typename t>
void date<t>::operator--()
{
	--dd;
	--mm;
	--yy;
}
template<typename t>												// postfix
void date<t>::operator++(t)
{
	dd++;
	mm++;
	yy++;
}
template<typename t>
void  date<t>::operator--(t)
{
	this->dd--;
	this->mm--;
	this->yy--;
}
template<typename t>
int date<t>::operator<(date& obj)
{
	bool dis;
	if (this->dd < obj.dd && this->mm < obj.mm && this->yy < obj.yy)
		// cout<<"Greater than equal to ";
		dis = true;
	else
		// cout<<"NOT Greater than equal to ";
		dis = false;
	return dis;
}
template<typename t>
int date<t>::operator>(date& obj)
{
	bool dis;
	if (this->dd > obj.dd && this->mm > obj.mm && this->yy > obj.yy)
		// cout<<"Greater than equal to ";
		dis = true;
	else
		// cout<<"NOT Greater than equal to ";
		dis = false;

}
template<typename t>
int date<t>::operator<=(date& obj)
{
	bool dis;
	if (this->dd <= obj.dd && this->mm <= obj.mm && this->yy <= obj.yy)
		// cout<<"Greater than equal to ";
		dis = true;
	else
		// cout<<"NOT Greater than equal to ";
		dis = false;
	return dis;
}
template<typename t>
int date<t>::operator>=(date& obj)
{
	bool dis;
	if (this->dd >= obj.dd && this->mm >= obj.mm && this->yy >= obj.yy)
		// cout<<"Greater than equal to ";
		dis = true;
	else
		// cout<<"NOT Greater than equal to ";
		dis = false;
	return dis;
}
template<typename t>
void date<t>::output()
{
	cout << this->dd << " / ";
	cout << this->mm << " / ";
	cout << this->yy << endl;
}
