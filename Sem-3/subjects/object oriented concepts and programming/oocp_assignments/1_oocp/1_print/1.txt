/* 1. Write a program to create class Student with studentâ€™s rollno, name and marks of three subjects (OOCP, DBMS and English)
and display the details of student with total marks of all subjects along with the percentage in proper format.
(Output should be in descending order of percentage).
*/
#include <iostream>
using namespace std;
class student
{
	int rollNo, marks[3];
	int total;
	char name[20];
public:
	int percent;
	void input();
	void process();
	void output();
};
int main()
{
	student s[10];
	int n;
	cout << "Enter no of student :";
	cin >> n;
	system("cls");
	for (int i = 0; i < n; i++)
	{
		s[i].input();
	}
	for (int i = 0; i < n; i++)
	{
		s[i].process();
	}
	//ORDER
	student temp;
	for (int i = 0; i < n; i++)
	{
		for (int j = i + 1; j < n; j++)
		{
			if (s[i].student::percent < s[j].student::percent)
			{
				temp = s[j];
				s[j] = s[i];
				s[i] = temp;
			}
		}
	}
	cout << "\n\nXXXXXXXXXXXXXX\n" << endl;
	for (int i = 0; i < n; i++)
	{
		s[i].output();
		cout << "\nPercent:" << s[i].percent << endl;
	}
	return 0;
}
void student::output()
{
	cout << "\nRollno:" << rollNo << "\nName:" << name << "\ntotal:" << total;
}
void student::process()
{
	percent = total / 3;
}
void student::input()
{
	int i;
	cout << "Enter roll.No :";
	cin >> rollNo;
	cout << "Name :";
	cin >> name;
	total = 0;
	for (i = 0; i < 3; i++)
	{
		cout << "Enter " << i + 1 << " subject marks : ";
		cin >> marks[i];
		total += marks[i];
		cout << "\n";
	}
	system("cls");
}
